类

js
class Person {
    name='';
    age=0;
    protected _height=0; // 子类可以访问
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    sayHello() {
        console.log('hello');
    }
    set height(height) {
        this._height = height;
    }
    get height() {
        return this._height;
    }
}

let p = new Person('zhangsan', 18);
p.height = 180;

// 重写的子类属性<父类，方法>父类
class Student extends Person {
    grade = 0;
    // 重写父类的构造函数，可以删除旧的参数
    constructor(name, age, grade) {
        super(name, age);
        this.grade = grade;
    }
}

Worker

js
// a.js
let worker = new Worker('b.js'); // 创建一个worker
worker.onmessage = function (e) {
    console.log(e.data);
    worker.terminate(); // 销毁worker
}
worker.onerror = function (e) {
    console.log(e.message);
    console.log(e.lineno);
    console.log(e.filename);
}
worker.postMessage('hello');

// b.js
self.importScripts('c.js');
self.onmessage = function (e) {
    console.log(e.data);
    self.postMessage('hello');
    self.close(); // 关闭worker
}

// 共享

let worker2 = new sharedWorker('b.js');

worker1.port.onmessage = function (e) {
}

worker threads

js
// a.js
import { Worker, parentPort } from 'worker_threads';
const worker = new Worker(__filename);
worker.on('message', (msg) => {
    console.log(msg);
    worker.terminate();
})
worker.postMessage('hello');
worker.on('error', (e) => {
    console.log(e.message);
})
worker.on('exit', (code) => {
    console.log('exit', code);
})

// b.js
parentPort.on('message', (msg) => {
    console.log(msg);
    parentPort.postMessage('hello');
    parentPort.close();
})
